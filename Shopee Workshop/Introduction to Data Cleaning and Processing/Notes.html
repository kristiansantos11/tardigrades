4) Extracting Numbers/Words from Strings¶

  • in the content column, there are data that has the format of '{"url":"https://seller.shopee.co.id/webview/rate_cards/?id=1246424278409256960&shop_id=196642601"}'
  • extract the id from this value 1246424278409256960
  • assign the id to the column, csat_id

# Create an empty list to store values 
rate_cards = []

# Repeat this step for each value in content
# ind is the index(row number), row is the row itself
for ind, row in df.iterrows():
    # To access the 'content' value  - row['content']
    
    # Check if inside the content value, there is the string 'rate_cards'
    if 'rate_cards' in row['content']:
        # apply the logic we used above
        rate = row['content'].split('id=')[1].split('&shop')[0]
    else:
        # if it doesnt have 'rate_cards in the content assign a blank string'
        rate = ''
        
    # push the value into the empty list
    rate_cards.append(rate)
    

# assign this list as a new column csat_id
df['csat_id'] = rate_cards    
    
# Check those rows with csat_data
df[df['csat_id'] != ''].loc[:,['message_id','content','csat_id']].head()

# Create a function

def assign_session(df):
    # first session starts with 0
    session_number = 1
    
    # create an empty list to store the session numbers
    session_list = []
    
    # go through each row
    for ind, row in df.iterrows():
        session_list.append(session_number)
        if row['conversation_status'] == 'closed':
            session_number = session_number + 1
    df['session'] = session_list
            
    return df
